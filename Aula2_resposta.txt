stb = QgsPointXY(-45.88441, -23.39755)

destino = QgsCoordinateReferenceSystem("EPSG:31983")
projeto = QgsProject.instance()

print("Distância:", jcr.distance(stb))
print("Distância em metros:", jcr_m.distance(stb_m))
